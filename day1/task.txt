1. /etc/passwd OPERATIONS

a) View /etc/passwd file
cat /etc/passwd

b) Print the 1st field from /etc/passwd file
awk -F: '{print $1}' /etc/passwd

c) Print all userids > 1000
awk -F: '$3 > 1000 {print $1}' /etc/passwd

d) Print the 2nd and 6th field
awk -F: '{print $2, $6}' /etc/passwd

e) Use command substitution to get user list and home directory
for i in $(awk -F: '{print $1":"$6}' /etc/passwd)
do
 echo $i
done

f) Change ownership of above home directory with user retrieved above
for i in $(awk -F: '{print $1":"$6}' /etc/passwd)
do
 user=$(echo $i | awk -F: '{print $1}')
 home=$(echo $i | awk -F: '{print $2}')
 sudo chown $user $home
done

g) Iterate above steps for all userid > 1000
for i in $(awk -F: '$3 > 1000 {print $1":"$6}' /etc/passwd)
do
 user=$(echo $i | awk -F: '{print $1}')
 home=$(echo $i | awk -F: '{print $2}')
 sudo chown $user $home
done


2. FILE OPERATIONS

a) Create files in current directory
touch abc.txt def.txt ghi.txt jkl.txt

b) Print list of files to move
ls *.txt

c) Segregate basename and extension of a file
for i in *.txt
do
 echo $i | awk -F. '{print "Basename:",$1,"Extension:",$2}'
done

d) Create folder using basename
for i in *.txt
do
 mkdir ${i%.*}
done

e) Move file to newly created folder
for i in *.txt
do
 mv $i ${i%.*}
done

f) Iterate above steps for all files
for i in *.txt
do
 echo $i | awk -F. '{print "Basename:",$1,"Extension:",$2}'
 mkdir ${i%.*}
 mv $i ${i%.*}
done


3. APPEND DATE TO LOG FILES

a) Create files
touch abc.log.1 def.log.1 ghi.log.1 jkl.log.1 mno.log.1

b) Print list of files
ls *.log.1

c) Segregate basename and extension
for i in *.log.1
do
 echo $i
done

d) Print date command
date +%d%m%Y

e) Append date to log filename
for i in *.log.1
do
 mv $i ${i%.log.1}-$(date +%d%m%Y).log
done
